            btn.addEventListener('click', (e) => { const idx = parseInt(btn.dataset.idx), delta = parseInt(btn.dataset.delta); if(!isNaN(idx) && quoteList[idx]) { let nq = quoteList[idx].quantity + delta; if(nq<1) nq=1; quoteList[idx].quantity=nq; saveQuote(); } e.stopPropagation(); });
        });
        document.querySelectorAll('.remove-item').forEach(btn => { btn.addEventListener('click', (e) => { const idx = parseInt(btn.dataset.idx); if(!isNaN(idx)) { quoteList.splice(idx,1); saveQuote(); } e.stopPropagation(); }); });
    }
    function addToQuote(product, qty=1) {
        const existing = quoteList.find(i=> i.id === product.id);
        if(existing) existing.quantity += qty;
        else quoteList.push({ id:product.id, name:product.name, price:product.price, moq:product.moq, img:product.img, quantity:qty });
        saveQuote();
        const toast = document.createElement('div'); toast.innerText = `✓ 已添加 ${product.name} 至询价清单`; toast.style.cssText = 'position:fixed; bottom:80px; left:20px; background:#1e1f25; color:white; padding:8px 18px; border-radius:40px; font-size:0.8rem; z-index:1001; backdrop-filter:blur(8px);'; document.body.appendChild(toast); setTimeout(()=>toast.remove(),2000);
    }

        const grid = document.getElementById('productGrid'); if(!grid) return;
        grid.innerHTML = '';

            grid.appendChild(card);
        });
        document.querySelectorAll('.add-to-quote').forEach(btn => { btn.addEventListener('click', (e) => { const pid = parseInt(btn.dataset.id); const prod = products.find(x=>x.id===pid); if(prod) addToQuote(prod,1); e.stopPropagation(); }); });
    }
